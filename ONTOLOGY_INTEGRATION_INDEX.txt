      test_id: &str,
      status: TestStatus,
      duration_ms: u64,
      coverage_delta: f32,
    ) {
      // Emit OCEL event
      ocel.add_event(OcelEvent {
        event_type: "Test.Executed",
        omap: { test: [test_id], workspace: [...] },
        attributes: {
          result: status.to_string(),
          duration_ms,
          coverage_delta,
        },
      });

      // Emit EARL assertion
      rdf_store.add(Triple {
        subject: test_result_uri,
        predicate: "earl:outcome".into(),
        object: match status { Pass => "earl:Pass", ... }
      });
    }
